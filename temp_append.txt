
      {/* Compilation Results Modal */}
      {showCompilationResults && (
        <CompilationResults
          result={compilationResult}
          onClose={closeCompilationResults}
        />
      )}
    </div>
  );
}
          {showTerminal && (
            <div className="border-t border-gray-700">
              <Terminal height={terminalHeight} />
            </div>
          )}
        </div>
      </div>
    </div>
  );
}
